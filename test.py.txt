#!/usr/bin/env python3
import docx
from htmldocx import HtmlToDocx

docout = docx.Document("myexemple.docx")
new_parser = HtmlToDocx()

html = '<p>normal<span style="color: rgb(161, 0, 0);">normalcolored</span><strong>bold</strong><strong style="color: rgb(178, 107, 0);">boldcolored</strong><strong><em>italic</em></strong><strong style="color: rgb(0, 102, 204);"><em>italiccolored</em></strong></p>'
temp = new_parser.parse_html_string(html)
out_para = docout.paragraphs[0]
out_para.text = "" #clear text but keep style
for run in temp.paragraphs[0].runs: #for each run, copy in myexemple.docx with properties
    output_run = out_para.add_run(run.text)
    # Run's font data
    output_run.style = run.style
    # Run's color data
    output_run.font.color.rgb = run.font.color.rgb #==None at the 4th iteration, must be #B26B00
    # Run's bold data
    output_run.bold = run.bold
    # Run's italic data
    output_run.italic = run.italic
    # Run's underline data
    output_run.underline = run.underline
    # Paragraph's alignment data
    out_para.paragraph_format.alignment = temp.paragraphs[0].paragraph_format.alignment


docout.save("myexemple.docx")
temp.save("temp.docx")
